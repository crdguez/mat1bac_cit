        \begin{questions}
        \question Resolver las siguientes inecuaciones:
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ \left|{3 x - 2}\right| - 0.25\leq 0 $  \begin{solution}  $ 0.583333333333333 \leq x \wedge x \leq 0.75 $  \end{solution} \part[1]  $ \left|{2 x + 6}\right| - 0.5\geq 0 $  \begin{solution}  $ \left(-2.75 \leq x \wedge x < \infty\right) \vee \left(x \leq -3.25 \wedge - \infty < x\right) $  \end{solution} \part[1]  $ \left|{x - 4}\right| - 2< 0 $  \begin{solution}  $ 2 < x \wedge x < 6 $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Calcula:
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ 2\sqrt {3125}  + 3\sqrt {20}  - 12\sqrt {45} $  \begin{solution}  $ 20 \sqrt{5} $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Realiza los desarrollos de los siguientes binomios:
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ ( {\frac{2}{\sqrt {2} } + \sqrt {2} } )^3 $  \begin{solution}  $ \frac{12 \sqrt{2}}{2} + 10 \sqrt{2} $  \end{solution} \part[1]  $ ( {2{x^2} - \frac{3}{x}} )^4 $  \begin{solution}  $ 16 x^{8} - 96 x^{5} + 216 x^{2} - \frac{216}{x} + \frac{81}{x^{4}} $  \end{solution} \part[1]  $ ( {\frac{x}{3} + \frac{3}{{{x^2}}}} )^5 $  \begin{solution}  $ \frac{x^{5}}{243} + \frac{5 x^{2}}{27} + \frac{10}{3 x} + \frac{30}{x^{4}} + \frac{135}{x^{7}} + \frac{243}{x^{10}} $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Dados los polinomios $ A(x)=2{x^3} - 3{x^2} + 2x - 1, \  B(x)=- {x^4} + {x^3} + x - 1, \  C(x)={x^4} - {x^2} + x + \frac{1}{3}$ halla:
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ x^2\cdot A(x) + 3x\cdot B(x) $  \begin{solution}  $ - x^{5} + 2 x^{3} + 2 x^{2} - 3 x $  \end{solution} \part[1]  $ A(x) - 3B(x) + 5C(x) $  \begin{solution}  $ 8 x^{4} - x^{3} - 8 x^{2} + 4 x + \frac{11}{3} $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Descomponer en factores
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ 2 x^{3} + 2 x^{2} - 12 x $  \begin{solution}  $ 2 x \left(x - 2\right) \left(x + 3\right) $  \end{solution}
        \end{parts}
        \end{multicols}
        
    \end{questions}
