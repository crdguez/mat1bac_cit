        \begin{questions}
        \question Resolver las siguientes inecuaciones:
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ \left|{2 x - 4}\right| - 2> 0 $  \begin{solution}  $ \left(-\infty < x \wedge x < 1\right) \vee \left(3 < x \wedge x < \infty\right) $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Calcula :
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ {2^{ - 5}} \cdot {( {\frac{1}{2}} )^{ - 2}} \cdot 2^7 $  \begin{solution}  $ 16 $  \end{solution} \part[1]  $ \frac{{{5^{ - 3}} \cdot {5^{ - 1}} \cdot {5^2}}}{{{5^0} + {5^6}}} $  \begin{solution}  $ \frac{1}{390650} $  \end{solution} \part[1]  $ \frac{{{4^{\frac{1}{2}}} \cdot {2^{ - 1}} \cdot {2^{\frac{9}{2}}}}}{{\sqrt {2} }} $  \begin{solution}  $ 16 $  \end{solution} \part[1]  $ {( {\frac{2}{5}} )^{ - 2}} \cdot {( {\frac{5}{2}} )^4} $  \begin{solution}  $ \frac{15625}{64} $  \end{solution} \part[1]  $ \frac{{\sqrt {2}  \cdot {{( {\sqrt {2} } )}^3} \cdot {{( {\sqrt {5} } )}^3}}}{{{{( {5\sqrt{2} } )}^2}}} $  \begin{solution}  $ \frac{2 \sqrt{5}}{5} $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Realiza los desarrollos de los siguientes binomios:
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ ( {1 + 3\sqrt{ 2} })^3 $  \begin{solution}  $ 55 + 63 \sqrt{2} $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Descomponer en factores
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ 6 x^{3} + 2 x^{2} - 24 x - 8 $  \begin{solution}  $ 6 \left(x - 2\right) \left(x + \frac{1}{3}\right) \left(x + 2\right) $  \end{solution} \part[1]  $ 2 x^{3} - 6 x^{2} - 26 x + 30 $  \begin{solution}  $ 2 \left(x - 5\right) \left(x - 1\right) \left(x + 3\right) $  \end{solution} \part[1]  $ 2 x^{5} - 8 x^{3} - 2 x^{2} + 8 $  \begin{solution}  $ 2 \left(x - 2\right) \left(x - 1\right) \left(x + 2\right) \left(x^{2} + x + 1\right) $  \end{solution}
        \end{parts}
        \end{multicols}
