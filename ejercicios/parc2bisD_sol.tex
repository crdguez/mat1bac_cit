        \begin{questions}
        \question Respecto de una base ortonormal tenemos dos vectores $\overrightarrow{u}$ y $\overrightarrow{v}$.
Calcular $\overrightarrow{u}\cdot\overrightarrow{v}$, $|\overrightarrow{u}| \ y \ |\overrightarrow{v}|$ 
y $\angle(\overrightarrow{u},\overrightarrow{v})$ siendo:
        \begin{multicols}{1} 
        \begin{parts} \part[2]  $ \overrightarrow{u}=(8, 2) \, \ \overrightarrow{v}=(-3, 4) $  \begin{solution}  $ \left [ -16, \quad \left [ 2 \sqrt{17}, \quad 5\right ], \quad 112.833654177918\right ] $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Si $\cos\alpha=-\frac{4}{5}\land \alpha \in III$, calcula (sin usar la calculadora, con identidades trigonométricas y reducción al primer cuadrante):
        \begin{multicols}{1} 
        \begin{parts} \part[1]  $ \tan\alpha $  \begin{solution}  $ \frac{3}{4} $  \end{solution} \part[1]  $ \sin\alpha $  \begin{solution}  $ - \frac{3}{5} $  \end{solution} \part[1]  $ \cos(\pi+\alpha) $  \begin{solution}  $ \frac{3}{5} $  \end{solution} \part[1]  $ \sin(2\alpha) $  \begin{solution}  $ \frac{24}{25} $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Resolver las siguientes ecuaciones:
        \begin{multicols}{1} 
        \begin{parts} \part[3]  $ \sin{2x}-\sin{x}=0 $  \begin{solution}  $ \left [ 0.0, \quad -180.0, \quad 180.0, \quad -300.0, \quad 300.0, \quad -60.0, \quad 60.0\right ] $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Calcula los ángulos y el perímetro de un triángulo ABC del que se conocen:
        \begin{multicols}{1} 
        \begin{parts} \part[3] Los lados a=60 cm y b=20 cm, y C=30º  \begin{solution}  $ \sqrt{- 1200 \sqrt{3} + 4000} = 43.83536278984522,\ 136.81321456798648º,\ 13.186785432013522º, \ 123.83536278984522 cm2 $  \end{solution}
        \end{parts}
        \end{multicols}
        \question Dado el siguiente número $z$, calcula $\frac{z+\overline{z}}{z-\overline{z}}$ dando el resultado en forma compleja binómica:
        \begin{multicols}{1} 
        \begin{parts} \part[2]  $ 16\sqrt{7}-2i $  \begin{solution}  $ 8 \sqrt{7} i $  \end{solution}
        \end{parts}
        \end{multicols}
        
    \end{questions}
