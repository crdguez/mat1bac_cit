        \begin{parts} \part[1]  $ \frac{{{x^2} - 1}}{{x + 3}} \ y  \ \frac{{{x^3} + 2{x^2} - x - 2}}{{{x^2} + 5x + 6}} $  \begin{solution}  $ \frac{\left(- x + \left(x^{3} + 2 x^{2}\right)\right) - 2}{\left(x^{2} + 5 x\right) + 6} - \frac{x^{2} - 1}{x + 3}=0 $  \end{solution} \part[1]  $ \frac{{{x^2} - 4}}{{{x^2} - 2x}}  \ y  \ \frac{{x + 2}}{x} $  \begin{solution}  $ - \frac{x^{2} - 4}{x^{2} - 2 x} + \frac{x + 2}{x}=0 $  \end{solution}
