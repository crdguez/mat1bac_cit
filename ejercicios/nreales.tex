    \question p9e3 - Resolver las siguientes inecuaciones:
        \begin{multicols}{3} 
        \begin{parts} \part[]  $ \left|{x - 2}\right| - 1< 0 $  \begin{solution}  $ 1 < x \wedge x < 3 $  \end{solution} \part[]  $ \left|{2 x + 3}\right| - 4< 0 $  \begin{solution}  $ - \frac{7}{2} < x \wedge x < \frac{1}{2} $  \end{solution} \part[]  $ \left|{x + 5}\right| - 2\leq 0 $  \begin{solution}  $ -7 \leq x \wedge x \leq -3 $  \end{solution} \part[]  $ \left|{3 x - 2}\right| - 0.5\leq 0 $  \begin{solution}  $ 0.5 \leq x \wedge x \leq 0.833333333333333 $  \end{solution} \part[]  $ \left|{x - 1}\right| - 2> 0 $  \begin{solution}  $ \left(-\infty < x \wedge x < -1\right) \vee \left(3 < x \wedge x < \infty\right) $  \end{solution} \part[]  $ \left|{x + 2}\right| - 5> 0 $  \begin{solution}  $ \left(-\infty < x \wedge x < -7\right) \vee \left(3 < x \wedge x < \infty\right) $  \end{solution} \part[]  $ \left|{2 x - 3}\right| - 1\geq 0 $  \begin{solution}  $ \left(2 \leq x \wedge x < \infty\right) \vee \left(x \leq 1 \wedge -\infty < x\right) $  \end{solution} \part[]  $ \left|{2 x + 1}\right| - 0.5\geq 0 $  \begin{solution}  $ \left(-0.25 \leq x \wedge x < \infty\right) \vee \left(x \leq -0.75 \wedge - \infty < x\right) $  \end{solution}
        \end{parts}
        \end{multicols}
